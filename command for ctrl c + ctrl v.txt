Файл із базовим набором команди для швидкого копіювання та вставлення*
мені поки складно кожного разу їх прописувати по новій, тому зробив таке
також можливо комусь буде цікаво поєксперементувати тому залишаю всі команди що знаю
# Навігація
cd monorepo\book-library\backend

# Запуск
npm run start:dev    // запуск з авто-перезавантаженням при змінах (використовуй завжди)
npm run start        // звичайний запуск без відстеження змін
npm run build        // збірка проєкту в папку dist/

# Nest CLI
nest generate resource ім'я    // створює module + controller + service + dto одразу
nest generate module ім'я      // тільки модуль
nest generate controller ім'я  // тільки контролер
nest generate service ім'я     // тільки сервіс

# Пакети
npm install                         // встановити всі залежності (після клонування)
npm i назва-пакету                  // встановити новий пакет
npm i назва-пакету --save-dev       // встановити пакет тільки для розробки

# Гит
git status                          // показує які файли змінені
git add .                           // додати всі зміни
git commit -m "опис змін"           // зберегти зміни з описом
git push                            // відправити на GitHub
git log --oneline                   // історія комітів коротко

# Тундер кліент, до books аналогичне
POST   /authors                     // створити (без id)
GET    /authors                     // всі автори
GET    /authors/1                   // один автор
PATCH  /authors/1                   // оновити
DELETE /authors/1                   // видалити
